```typescript
function getTiedClusters(standings: TeamStanding[]): TeamStanding[][] {
  const clusters: TeamStanding[][] = [];
  let i = 0;
  while (i < standings.length) {
    let j = i + 1;
    while (j < standings.length && standings[j].points === standings[i].points) j++;
    if (j - i > 1) clusters.push(standings.slice(i, j));
    i = j;
  }
  return clusters;
}
```

```typescript
function findNextUnscoredId(allRaceIds: string[], scoreMap: Map<string, Score>, afterId: string): string | null {
  const idx = allRaceIds.indexOf(afterId);
  for (let i = idx + 1; i < allRaceIds.length; i++) {
    if (!scoreMap.has(allRaceIds[i])) return allRaceIds[i];
  }
  return null;
}
```

const handleScore = useCallback((result: { raceId: string; homeSlot: number; awaySlot: number; homeOutcome: RaceOutcome; awayOutcome: RaceOutcome }) => {

  // Compute next unscored synchronously before React re-renders (Pitfall 2)

}, [discipline, recordResult, scoreMap, allRaceIds]);
